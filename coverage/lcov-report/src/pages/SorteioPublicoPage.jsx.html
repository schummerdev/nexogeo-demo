const MediaWithFallback = ({ src, fallbacks = [], alt, className, style, ...props }) => {
  const [currentSrc, setCurrentSrc] = useState(src);
  const [fallbackIndex, setFallbackIndex] = useState(-1);
  const [hasErrored, setHasErrored] = useState(false);
 
  const handleError = () => {
    if (fallbackIndex < fallbacks.length - 1) {
      const nextIndex = fallbackIndex + 1;
      setFallbackIndex(nextIndex);
      setCurrentSrc(fallbacks[nextIndex]);
      console.log(` Tentando fallback ${nextIndex + 1}:`, fallbacks[nextIndex]);
    } else {
      setHasErrored(true);
      console.log(' Todos os fallbacks falharam, ocultando mdia');
    }
  };
 
  if (hasErrored) {
    return (
      <div
        className={className}
        style={{
          ...style,
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          backgroundColor: '#f0f0f0',
          color: '#666',
          fontSize: '14px',
          padding: '20px'
        }}
      >
         Mdia no disponvel
      </div>
    );
  }
 
  return (
    <img
      src={currentSrc}
      alt={alt}
      className={className}
      style={style}
      onLoad={() => console.log(` Mdia carregada: ${currentSrc}`)}
      onError={handleError}
      {...props}
    />
  );
};
